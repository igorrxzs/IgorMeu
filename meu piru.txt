/* Em uma festa, homens pagam 20 reais de entrada e as mulheres pagam 17. 
Crie um programa que pergunta quantos homens e quantas mulheres participaram da festa e calcula o valor total 
arrecadado com as entradas.*/

#include <stdio.h>

int main() {
    int homens, mulheres;
    float total;

    printf("Quantos homens participaram da festa?: ");
    scanf("%d", &homens);
    printf("Quantas mulheres participaram da festa?: ");
    scanf("%d", &mulheres);
    total = (homens * 20) + (mulheres * 17);
    printf("O valor total arrecadado foi: R$ %.2f\n", total);

    return 0;
}
